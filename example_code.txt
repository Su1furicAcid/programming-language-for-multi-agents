agent Summarizer:
    output:
        summary: str
    model: "gpt-4o"
    text = "syntax error"
    chat make_summary : """
    Make a summary of the text below.
    Text: {text}
    Summary: ${summary}
    """

agent Analyzer:
    input:
        summary
    model: "gpt-4o"
    chat : """
    Analyze the text.
    Text: {summary}
    """

connect:
    summary_line : str 
        Summarizer.output.summary -> Analyzer.input.summary
